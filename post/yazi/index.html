# ./config/plugins/open-folder.yazi/main.lua
--- @sync entry

return {
 entry = function()
  local h = cx.active.current.hovered
  if not h then
   return ya.notify { title = "Open Folder", content = "No file selected", timeout = 2, level = "warn" }
  end

  local path
  if h.cha.is_dir then
   path = tostring(h.url)
  else
   path = tostring(h.url:parent())
  end

  -- io.popen runs asynchronously and won't block Yazi
  io.popen('explorer "' .. path .. '"'):close()
 end,
}
